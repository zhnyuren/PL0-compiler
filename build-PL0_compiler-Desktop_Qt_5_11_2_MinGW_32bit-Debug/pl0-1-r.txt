const z=0, tMp=1;
var head,foot,cock,rabbit,n;
begin
   n:=z;
   read(head,foot);
   cock:=0;
   while cock<=head do
   begin
      rabbit:=head-cock;
      if cock*2+rabbit*4 = foot then
      begin
          write(cock,rabbit);
          n:=n+1
      end;
      cock:=cock+1
   end;
   if n=0 then write(tMp)
end.
